rescue cd on usb thumb drive

# df -Th
/dev/sdd1     vfat    3.8G  4.0K  3.8G   1% /media/KINGSTON


dmesg

--
-- mount cdrom
--
mkdir foo
mount -o loop systemrescuecd-x86-1.3.4.iso foo

--
-- prepare usb (may not be needed for a new usb???)
--
cfdisk /dev/sdd
  make it bootable

mkfs.vfat -F 32 -n SYSREC /dev/sdd1 
dd if=/usr/lib/syslinux/mbr.bin of=/dev/sdd
sync


-- 
-- copy files over
--
mount /dev/sdd1 usb

cp -af foo/* usb/
cd usb/isolinux/
mv isolinux.cfg syslinux.cfg
cd ..
mv isolinux/ syslinux


--
-- make it bootable (again???)
--
syslinux /dev/sdd1 
sync
