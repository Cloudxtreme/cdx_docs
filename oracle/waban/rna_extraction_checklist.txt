select functional_step_name func_name, p.procedure_name 
from Step_Template st,checklist c,functional_step f, procedure_template p 
where st.procedure_template_id=c.procedure_template_id 
and p.procedure_template_id=c.procedure_template_id 
and f.functional_step_id (+)=st.functional_step_id 
and c.checklist_id=17992 
order by number_in_sequence
