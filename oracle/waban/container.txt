SELECT   c.container_id, c.purpose,
	       cm.container_rows
	    || ' Rows * '
	    || cm.container_columns
	    || CASE
		  WHEN UPPER (cm.TYPE) = 'GEL'
		     THEN ' Lanes'
		  ELSE ' Columns'
	       END AS layout,
	    limspackage.freezer_position (c.container_id) container_location,
	    limspackage.containerfreespace (c.container_id) freesp,
	    lims_assay_container.fn_get_assay_plate_checklistid
					     (c.container_id,
					      'C'
					     ) AS checklist_id,
	    lims_assay_container.fn_get_assay_plate_checklistid
					   (c.container_id,
					    'P'
					   ) AS procedure_name,
	    Status, Comments, ' ' last_col, c.freezer_id, c.container_map_id,
	    c.compartment, c.shelf, c.rack, c.LOCATION, c.description,
	    c.checkout_by, c.date_checkout, c.discarded, c.created_by,
	    c.date_created, c.modified_by, c.date_modified, c.subsidiary_id,
	    c.container_condition, cm.TYPE
FROM container c, container_map cm
WHERE c.container_map_id = cm.container_map_id
	AND NVL (c.discarded, 0) = 0
ORDER BY container_id
