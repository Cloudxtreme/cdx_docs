
select cs.checklist_id, cs.checklist_col, cs.sample_id, cs.orig_sample_id, cs.status, 
		s1.sample_tube_id currTube,a1.accession_number currAcc,
		limspackage.container_position(s1.sample_id) location,
		s2.sample_tube_id origTube,a2.accession_number origAcc 
from checklist_sample cs,sample s1,sample s2,accession a1,accession a2 
where s1.Sample_ID=cs.Sample_ID 
and s1.accession_id=a1.accession_id 
and s2.Sample_ID=cs.orig_Sample_ID 
and s2.accession_id=a2.accession_id 
and checklist_ID=19352 order by checklist_col

select obj#,type#,ctime,mtime,stime,status,dataobj#,flags,oid$, spare1, spare2 from obj$ where owner#=:1 and name=:2 and namespace=:3 and remoteowner is null and linkname is null and subname is null

select * from checklist_content where checklist_ID=19352 order by checklist_row, checklist_col
select count(*) row_count from (select * from checklist where date_completed is null and checklist_id=19352)
select ptr.* from checklist chk,procedure_template_result ptr where chk.procedure_template_id = ptr.procedure_template_id and chk.checklist_id = 19352 order by ptr.number_in_sequence


select distinct c.checklist_name || '/' || c.checklist_id batch_id, cc.checklist_col, cc.checklist_row, cc.entered_value
				from checklist_sample cs, checklist c, checklist_content cc, procedure_template pt,
					sample s, accession a
				where cs.checklist_id = c.checklist_id
				and c.checklist_id = cc.checklist_id
				and c.procedure_template_id = pt.procedure_template_id
					and cs.sample_id = s.sample_id
					and a.accession_id = s.accession_id
				and pt.procedure_name = 'Bubble Removal/Volume Check'
				and cs.status = 'Passed'
				and (cc.checklist_col > 0 and cc.checklist_row in (1, 4, 5))
				and a.accession_number in ('TRF58109')
				order by batch_id, cc.checklist_col, cc.checklist_row
