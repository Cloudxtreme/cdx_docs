from study_variable_fld svf, protocol_defined_variable pdv, variable var 
where svf.protocol_defined_variable_id = pdv.protocol_defined_variable_id 
and pdv.variable_id = var.variable_id 
and var.variable_name in ('Min Sample Volume', 'Max Sample Volume')
and svf.study_protocol_id in (2, 53, 117);
