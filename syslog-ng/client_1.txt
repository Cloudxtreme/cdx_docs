

options { sync (0);
  time_reopen (10);
  log_fifo_size (1000);
  long_hostnames (off);
  use_dns (no);
  use_fqdn (no);
  create_dirs (no);
  keep_hostname (yes);
};

source sys { 
  file ("/proc/kmsg" log_prefix("kernel: ")); 
  unix-stream ("/dev/log"); 
  internal(); 
};

destination loghost { udp("syslog" port(514)); };

log { source(sys); destination(loghost); };
