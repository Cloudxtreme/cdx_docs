select pc.patient_case_id, ad.appeal_detail_id
from pc_fact pc
left join ad_fact ad
  on pc.patient_case_id = ad.patient_case_id

select pc.patient_case_id, pad.payment_adjustment_detail_id
from pc_fact pc
left join pad_fact pad
  on pc.patient_case_id = pad.patient_case_id


select pc.patient_case_id, ad.appeal_detail_id, pad.payment_adjustment_detail_id
from pc_fact pc
left join ad_fact ad
  on pc.patient_case_id = ad.patient_case_id
left join pad_fact pad
  on pc.patient_case_id = pad.patient_case_id


select pc.patient_case_id, ad.appeal_detail_id, pad.payment_adjustment_detail_id, plan.plan_id
from pc_fact pc
left join ad_fact ad
  on pc.patient_case_id = ad.patient_case_id
left join pad_fact pad
  on pc.patient_case_id = pad.patient_case_id
	left join plan_fact plan
	on ad.plan_id = plan.plan_id


select pc.patient_case_id, pad.payment_adjustment_detail_id
from pc_fact pc
right join pad_fact pad
  on pc.patient_case_id = pad.patient_case_id
	right join plan_fact plan 
	on pad.plan_id = plan.plan_id
