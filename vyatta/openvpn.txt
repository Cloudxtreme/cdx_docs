

-- client/server w/ tls
set interfaces openvpn vtun0
set interfaces openvpn vtun0 mode server
set interfaces openvpn vtun0 server subnet 192.168.1.0/24
set interfaces openvpn vtun0 tls ca-cert-file /root/ca.crt
set interfaces openvpn vtun0 tls cert-file /root/V1.crt
set interfaces openvpn vtun0 tls crl-file /root/crl.pem
set interfaces openvpn vtun0 tls dh-file /root/dh1024.pem
set interfaces openvpn vtun0 tls key-file /root/V1.key
commit

show interfaces openvpn vtun0
